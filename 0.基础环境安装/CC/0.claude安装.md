# 1. 更新包列表并安装 curl（如果没有的话）
apt-get update
apt-get install -y curl

# 2. 添加 NodeSource 官方源并安装 Node.js 20.x（包含 npm）
curl -fsSL https://deb.nodesource.com/setup_20.x | bash -
apt-get install -y nodejs

# 3. 验证安装
node -v   # 应该显示 v20.x
npm -v    # 应该显示 10.x

# 4. 安装 Claude Code
npm install -g @anthropic-ai/claude-code (比较慢，几分钟)

# 5. 验证 Claude Code 安装
claude --version   # 或者直接运行 claude

# 6.添加插件超市
  claude market add obra/superpowers-marketplace
  claude market add jarrodwatts/claude-hud

  claude market add ananddtyagi/cc-marketplace 
  claude market add Dev-GOM/claude-code-marketplace 

# 7.安装插件
  claude plugins add superpowers@claude-plugins-official
  claude plugins add claude-hud@claude-hud
